const

  ArrSize = 13;

type

  TArray = array [0..ArrSize] of integer;

var

  Arr: TArray;

procedure JumpSearch(var arr: TArray; element:integer);

begin

  var high := ArrSize;

  var jump:integer := floor(sqrt(high));

  var left:integer := 0;

  var right:integer := 0;

  var i:integer := 0;

while (left < high) and (Arr[left] <= element) do

   begin

    right := min(high-1, left+jump);

      if (Arr[left] <= element) and (Arr[right] >= element) then

        break;

    left := left + jump;

   end;

    if (left >= high) or (Arr[left] > element) then

        exit;

  right := min(high-1, right);

  i := left;

  while (i <= right) and (Arr[i] <=  element) do

    begin

      if  Arr[i] = element then

      begin

        print('Число найдено под индексом: ',Arr[i]);

        exit;

      end;

      i:=i+1;

    end;

end;

 

begin

  for var i := 0 to ArrSize do

    Arr[i] := i;

  JumpSearch(Arr, 7);

end.
