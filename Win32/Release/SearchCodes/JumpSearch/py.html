from math import sqrt

 

 

def jump_search(lis, item):

    high = len(lis)

    jump = int(sqrt(high))

    left, right = 0, 0

    while left < high and lis[left] <= item:

        right = min(high - 1, left + jump)

        if lis[left] <= item and lis[right] >= item:

            break

        left += jump

    if left >= high or lis[left] > item:

        return -1

    right = min(high-1, right)

    i = left

    while i <= right and lis[i] <= item:

        if lis[i] == item:

            return i

        i += 1

    return None

 

 

lis = []

for i in range(10000):

    lis.append(i)

print(jump_search(lis, 9992))
