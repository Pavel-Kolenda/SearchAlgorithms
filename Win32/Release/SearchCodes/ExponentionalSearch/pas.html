const

  ArrSize = 13;

type

  TArray = array [0..ArrSize] of integer;

var

  Arr: TArray;

 

procedure ExBinarySearch(var arr: TArray; left, right, element:integer);

begin

if left > right then

  exit;

var mid := left+(right-left) div 2;

if element = Arr[mid] then

  begin

    println('Число найдено под индексом', Arr[mid]);

    exit

  end

else if Arr[mid] > element  then

    ExBinarySearch(Arr, left, mid - 1, element)

else

    ExBinarySearch(Arr, mid + 1, right, element);

end;

 

procedure ExponentionalSearch(var arr: TArray; element:integer);

begin

  var bound := 1;

  var high := ArrSize;

  while (bound < high) and (Arr[bound] < element) do

    begin

    bound := bound * 2;

    end;

    ExBinarySearch(Arr, bound div 2, min(bound, high), element)

end;

 

begin

  for var i := 0 to ArrSize do

    Arr[i] := i;

  ExponentionalSearch(Arr, 13);

end.
