def exbinary_search(lis, left, right, element):

    if left > right:

        return -1

    mid = (left + right) // 2

    if element == lis[mid]:

        return mid

    elif element < lis[mid]:

        return exbinary_search(lis, left, mid - 1, element)

    else:

        return exbinary_search(lis, mid + 1, right, element)

 

 

def exponential_search(lis, element):

    bound = 1

    while bound < len(lis) and lis[bound] < element:

        bound *= 2

    return exbinary_search(lis, bound // 2, min(bound, len(lis) - 1), element)
