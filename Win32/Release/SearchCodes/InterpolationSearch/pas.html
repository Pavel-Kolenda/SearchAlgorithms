const

  ArrSize = 13;

type

  TArray = array [0..ArrSize] of integer;

var

  Arr: TArray;

 

procedure InterpolationSearch(var arr: TArray; element:integer);

begin

var low := 0;

var high := ArrSize;

var index: integer;

 

var a := Arr[low];

var b := Arr[high];

 

while (low <= high) and (Arr[low] <= element) and (Arr[high] >= element) do

  begin

  index := low + floor((((high-low)/ (b - a)) * (element - a)));

  if Arr[index] = element then

    begin

      println('Число найдено под индексом', Arr[index]);

      exit;

    end;

  if Arr[index] < element  then

      low := index + 1

  else

      high := index - 1;

  end;

println('Число не найдено') 

end;

 

begin

  for var i := 0 to ArrSize do

    Arr[i] := i;

  InterpolationSearch(Arr, 1);

end.
